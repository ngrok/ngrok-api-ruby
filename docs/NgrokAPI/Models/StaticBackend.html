def update(
  description: nil,
  metadata: nil,
  address: "",
  tls: nil
)
  @description = description if description
  @metadata = metadata if metadata
  @address = address if address
  @tls = tls if tls
  @client.update(
    id: @id,
    description: description,
    metadata: metadata,
    address: address,
    tls: tls
  )
end
