def get_next
  item = @page.attrs[@list_property][@n]
  raise "None" if item.nil?
  self.n += 1
  item
rescue
  if @page.next_page_uri
    res = @client.list(danger: @danger, url: @page.next_page_uri)
    self.n = 0
    self.page = res
    get_next
  end
end
